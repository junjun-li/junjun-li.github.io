            # 逻辑"与"运算符

            public class Demo {
  public static void main(String[] args) {
    // &一个运算符
    int n = 3;
    boolean b = (3 > 7) & ((n++) < 2); // 问: b=? n=?
    // 输出 b = false; n = 4;
    
    // 两个 && 又叫短路运算符 如果左边的计算为false, 右边的就不执行了
    boolean b = (3 > 7) && ((n++) < 2);
    // 输出b = false; n = 3; 因为左边的是false, 所有右边的代码不会执行了
  }
}

            # 逻辑"或"运算符

            public class Demo {
  public static void main(String[] args) {
    // |运算符
    int n = 3;
    boolean b = (3 > 7) | ((n++) < 2); // 问: b=? n=?
    // b = true, n = 4
    boolean b = (3 > 7) || ((n++) < 2); // 问: b=? n=?
    // b = true, n = 3
    // "||"运算符是: 如果左边计算后的操作数为true，右边则不再执行，返回true。
    //  "|"运算符是：前后两个操作数都会进行计算。
  }
}
